CREATE TABLE `blogv1`.`posts` (
  `id` INT NOT NULL AUTO_INCREMENT COMMENT '',
  `user_id` VARCHAR(45) NULL DEFAULT 0 COMMENT '',
  `kategori_id` VARCHAR(45) NULL DEFAULT 0 COMMENT '',
  `title` VARCHAR(45) NULL COMMENT '',
  `tags` VARCHAR(50) NULL COMMENT '',
  `post_preview` VARCHAR(250) NULL COMMENT '',
  `post` TEXT(999) NULL COMMENT '',
  PRIMARY KEY (`id`)  COMMENT '')
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8
COLLATE = utf8_turkish_ci;
